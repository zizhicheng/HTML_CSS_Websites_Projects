Arrays:

  1. var pets = ["dog","cat","rabbit"]
    var pets = new Array(10) --- An array with 10 enpty elements
  2. [] --- pets[2] --- rabbit
    ------ pets[2] = "bird" ----- pets will be ["dog","cat","bird"]
  3. length --- pets.length --- 3
  4. unshift() --- pets.unshift("bird") --- adding elements to the beginning of the array --- pets will be ["bird","dog","cat","rabbit"]
  5. push() --- pets.push("bird") --- adding extra element to the end of the array --- pets will be ["dog","cat","rabbit","bird"]
  6. shift() --- pets.shift() --- remove the first element from the array --- pets will be [cat","rabbit"]
  7. pop() --- pets.pop() --- remove the last element of the array --- pets will be ["dog","cat"]
  8. concat() --- array1.concat(array2) --- conbine two arrays together --- final array will be [array1,array2]
  9. join() --- pets.join(" and ") --- join all elements in pets with and between --- dog and cat and rabbit
    ---------- pets.join() --- join all elements in pets --- dog,cat,rabbit
